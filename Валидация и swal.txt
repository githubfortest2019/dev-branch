var $fill_form = $('#fill_form');
    
    $('#button_send').on('click', function() {
        $fill_form.validate({
            ignore: "",
            rules: {
                surname: {
                    required: true
                },
                name: {
                    required: true
                },
                patronymic: {
                    required: true
                },
                email: {
                    required: true,
                    email: true
                }
            },
            messages: {
                surname: {
                    required: "Данное поле не может быть пустым"
                },
                name: {
                    required: "Данное поле не может быть пустым"
                },
                patronymic: {
                    required: "Данное поле не может быть пустым"
                },
                email: {
                    required: "Данное поле не может быть пустым",
                    email: "E-mail должен быть формата user@domain.com"
                }
            }
        });
        
        if ($fill_form.valid()) {
            swal({
                title: "Поздравляем!",
                text: "Заявка на регистрацию успешна отправлена",
                type: "success",
                confirmButtonColor: "#3c8dbc",
                allowOutsideClick: true,
                confirmButtonText: "ОК",
            })
            .then(function() {
                location.reload();
            }).catch(swal.noop);
        }
    });
